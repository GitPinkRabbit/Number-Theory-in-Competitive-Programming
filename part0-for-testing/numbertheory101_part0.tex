  \begin{theorem}[可选定理名]
    不存在最大的素数。
  \end{theorem}
  
  \begin{proof}[可选证明附加文字]
    \begin{enumerate}
      \item<1-> 假设 $p$ 为最大的素数。
      \item<3-> Let $q$ be the product of the first $p$ numbers. 以下跳一步。
      \item<2-> But $q + 1$ is greater than $1$, thus divisible by some prime
      number not in the first $p$ numbers.\qedhere
    \end{enumerate}
  \end{proof}
